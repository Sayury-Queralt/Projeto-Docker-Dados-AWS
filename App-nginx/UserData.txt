#!/bin/bash
# 1. Definições
REPO_URL="https://github.com/Sayury-Queralt/Projeto-Docker-Dados-AWS.git"
# Onde o repositório inteiro vai ficar
BASE_DIR="/home/ec2-user/Projeto-Docker-Dados-AWS"
# Onde os arquivos do Docker/Nginx realmente estão (dentro do repo)
APP_DIR="$BASE_DIR/App-nginx"

exec > /var/log/user-data.log 2>&1

# 2. Clonar o repositório na pasta base
if [ ! -d "$BASE_DIR" ]; then
    git clone $REPO_URL $BASE_DIR
else
    cd $BASE_DIR && git pull origin master
fi

# 3. Ajustar permissões para o ec2-user
chown -R ec2-user:ec2-user $BASE_DIR

# 4. ENTRAR na pasta onde o Docker Compose e o Dockerfile estão
cd $APP_DIR

# 5. Subir o ambiente
docker compose pull
docker compose up -d
